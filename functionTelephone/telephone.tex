% 1/(1+exp(-(a*x+b)))
$$
f_1(x) = \frac{1}{1+e^{-(ax+b)}}
$$
\clearpage

\vspace{-2in}
%  exp(-((x-a)/b)^2/2)/(b*sqrt(2*pi))
$$
f_2(x) = \frac{1}{b \sqrt{2\pi}}e^{-\frac{(x-a)^2}{2b^2}}
$$
\clearpage

\vspace{-2in}
%  1-exp(-(x/b)^a)
$$
f_3(x) = 1-e^{-(x/b)^a}
$$
\clearpage

\vspace{-2in}
%  (a/b)*(x/b)^(a-1)*exp(-(x/b)^a)
$$
f_4(x) = \frac{a}{b}\left(\frac{x}{b}\right)^{a-1} e^{-(x/b)^a}
$$
\clearpage

\vspace{-2in}
%  a*b*x^(a-1)*(1-x^a)^(b-1) 
$$
f_5(x) =abx^{a-1}(1-x^a)^{b-1}
$$
\clearpage

\vspace{-2in}
%   1-(1-x^a)^b
$$
f_6(x) =  1-(1-x^a)^b
$$
\clearpage

\vspace{-2in}
%   a*(x-b)^2
$$
f_7(x) = a(x-b)^2
$$

\clearpage

\vspace{-2in}
%   a*b*exp(a*x)*exp(b)*exp(-b*exp(a*x))
$$
f_8(x) = abe^{ax}e^{b}e^{-b e^{ax}}
$$
\clearpage

\vspace{-2in}
%   a*x/(1+x/b)
$$
f_9(x) = \frac{ax}{1+x/b}
$$
\clearpage

\vspace{-2in}
%   x*exp(a*(1-x/b))
$$
f_{10} = x e^{a(1-x/b)}
$$
\clearpage

\vspace{-2in}
%   a*x/(a*x - (1-a)*b)
$$
f_{11}=\frac{ax}{ax-(1-a)b}
$$
\clearpage

\vspace{-2in}
%   a*x*(1-x/b)
$$
f_{12}=ax(1-x/b)
$$
